// lambda的递归
int sum(int n)
{
    std::function<int(int, int)> sum_inner;
    sum_inner = [&sum_inner, &n](int cur, int next) -> int
    {
        if (next > n)
        {
            return cur;
        }
        return sum_inner(cur + next, next + 1);
    };
    return sum_inner(0, 1);
}
