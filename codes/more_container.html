class A {
public:
	A(int i) :m_i(i) {}
	int m_i;
	bool operator < (const A& that) const //用以支持 std::map
	{
		return m_i < that.m_i;
	}
	bool operator == (const A& that) const //用以支持 std::unordered_map
	{
		return m_i == that.m_i;
	}
};
 
namespace std { // 定义hash functor的方式一
	template<>
	struct hash<A>
	{
		[[nodiscard]] size_t operator()(const A& v) const noexcept
		{
			return v.m_i;
		}
	};
}
// 使用
std::unordered_map<A, int> m2{ {1,1},{2,2},{3,3} }; // 方式一
 
struct hash_A // 定义hash functor的方式二
{
	[[nodiscard]] size_t operator()(const A& v) const noexcept
	{
		return v.m_i;
	}
};
// 使用
std::unordered_map<A, int, hash_A> m2{ {1,1},{2,2},{3,3} }; // 方式二

template<size_t ...>
struct index_helper {};
 
template<size_t N, size_t ...S>
struct index_wrapper : index_wrapper<N - 1, N - 1, S...> {};
 
template<size_t ...S>
struct index_wrapper<0, S...> { typedef index_helper<S...> type; };
 
 
template<typename Func, typename Tuple, size_t... indexes>
auto apply_impl(Func&& func, Tuple&& args, index_helper<indexes...>)
{
	return std::forward<Func>(func)(std::get<indexes>(std::forward<Tuple>(args))...);
}
 
template<typename Func, typename... Args> // 重载1
auto apply(Func&& func, std::tuple<Args&&...> arg) -> typename std::result_of<Func(Args...)>::type
{
	typedef std::tuple<Args&&...>                Tuple;
	typedef index_wrapper<sizeof...(Args)>::type Indexes;
	return apply_impl(std::forward<Func>(func), std::forward<Tuple>(arg), Indexes{});
}
 
template<typename Func, typename... Args> // 重载2
auto apply(Func&& func, std::tuple<Args...> arg) -> typename std::result_of<Func(Args...)>::type
{
	typedef std::tuple<Args...>                  Tuple;
	typedef index_wrapper<sizeof...(Args)>::type Indexes;
	return apply_impl(std::forward<Func>(func), std::forward<Tuple>(arg), Indexes{});
}
 
void test()
{
	auto sum = [](int a, double b, char c) { return a + b + c; };
 
	auto x1 = apply(sum, std::forward_as_tuple(10, 3.14, 'c')); // 调用重载1
	std::cout << x1 << std::endl;
	auto x2 = apply(sum, std::make_tuple(10, 3.14, 'c')); // 调用重载2
	std::cout << x2 << std::endl;
}
