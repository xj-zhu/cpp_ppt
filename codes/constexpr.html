// 声明constexpr表达式
constexpr int num = 10;
constexpr size_t M = 1024 * 1024;
 
// 声明返回constexpr的函数
constexpr int accumulation(int num)
{
    int ret = 0;
    for (int i = 1; i <= num; ++i)
    {
        ret += i;
    }
    return ret;
}
constexpr auto x = accumulation(100);
 
// 声明返回constexpr的函数的错误用例
#include <time.h>
constexpr time_t now_ts()
{
    return time(nullptr);
}
constexpr auto n = now_ts();
 
//编译错误：
error C3615 : constexpr 函数 "now_ts" 不会生成常数表达式
