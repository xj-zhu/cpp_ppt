class TimerMgr
{
public:
    typedef std::chrono::steady_clock::time_point TimePoint;
    typedef std::function<void(void)> TimerFunc; //定时器回调类型
    typedef std::shared_ptr<TimerFunc> SharedTimerFunc; //定时器回调指针类型
public:
    template <typename Duration
        , typename std::enable_if<std::is_convertible<typename std::decay<Duration>::type, std::chrono::steady_clock::duration>::value, int>::type = 0>
    inline int SetTimer(Duration&& duration, SharedTimerFunc func)
    {
        int id = _GenSeqId();
        m_regq[id] = func;
        m_trgq.insert(std::make_pair(std::chrono::steady_clock::now() + std::forward<Duration>(duration), std::make_pair(id, func)));
 
        std::this_thread::sleep_for(duration);
        return id;
    }
    template <typename Duration
        , typename std::enable_if<std::is_convertible<typename std::decay<Duration>::type, int>::value, int>::type = 0>
    inline int SetTimer(Duration&& duration, SharedTimerFunc func)
    {
        return SetTimer(std::chrono::seconds(duration), func);
    }
    inline void KillTimer(int id)
    {
        m_regq.erase(id);
    }
    void Run()
    {
        while (true)
        {
            TimePoint tp = std::chrono::steady_clock::now();
            while (true)
            {
                if (m_trgq.empty())
                {
                    break;
                }
                const auto it = m_trgq.begin();
                if (it->first > tp)
                {
                    break;
                }
                auto func = it->second.second.lock();
                if (func)
                {
                    (*func)(); //调用触发函数
                    m_regq.erase(it->second.first);
                }
                m_trgq.erase(it);
            }
        }
    }
 
private:
    int _GenSeqId() { static int id = 0; return ++id; }
 
private:
    typedef std::unordered_map<int, std::shared_ptr<TimerFunc>> TimerRegQ; //定时器注册队列
    typedef std::multimap<TimePoint, std::pair<int, std::weak_ptr<TimerFunc>>> TimerTrigerQ; //定时器触发队列
    TimerRegQ m_regq;
    TimerTrigerQ m_trgq;
};
