#include <string>
 
class MyStr : public std::string
{
public:
    using std::string::string; // 继承构造函数
    MyStr(int n) : MyStr(n, 'x') {} // 代理构造函数
    bool operator==(const MyStr& s) = default;
    MyStr(const MyStr& that) = delete;
 
    virtual void Clear()
    {
        std::cout << "mystr clear()" << std::endl;
        clear();
    }
};
 
class MyStr1 : public MyStr
{
public:
    using MyStr::MyStr;
    void Clear() override final
    {
        std::cout << "mystr1 clear()" << std::endl;
        MyStr::Clear();
    }
};
 
class MyStr2 : public MyStr1
{
public:
    using MyStr1::MyStr1;
    void Clear() override // 编译报错
    {
        std::cout << "mystr1 clear()" << std::endl;
        MyStr1::Clear();
    }
};
 
auto s1 = MyStr(10, 'x');
std::cout << "s1 is " << s1 << "." << std::endl;
auto s2 = MyStr(10);
std::cout << "s2 is " << s2 << "." << std::endl;
 
auto s3 = std::move(s1);
std::cout << "s1 is " << s1 << "." << std::endl;
std::cout << "s3 is " << s3 << "." << std::endl;
s3.Clear();
std::cout << "s3 is " << s3 << "." << std::endl;
auto s4 = s2; // 编译报错
auto s5 = MyStr1("hello s5");
std::cout << "s5 is " << s5 << "." << std::endl;
s5.Clear();
std::cout << "s5 is " << s5 << "." << std::endl;
