#include <iostream>
 
class Data
{
public:
    Data() // 无参构造函数
    {
        std::cout << "standard constructor" << std::endl;
    }
    Data(const std::string& num) : m_data{ num } // 有参构造函数
    {
        std::cout << "within args constructor" << std::endl;
    }
    ~Data() // 析构函数
    {
        std::cout << "denstructor" << std::endl;
    }
public:
    Data(const Data& other) : m_data{ other.m_data } // 拷贝构造函数
    {
        std::cout << "copy constructor" << std::endl;
    }
    Data(Data&& other) :m_data{ std::move(other.m_data) } // 移动构造函数
    {
        std::cout << "move constructor" << std::endl;
    }
public:
    Data& operator=(const Data& that) // 拷贝赋值函数
    {
        std::cout << "copy operator =" << std::endl;
        if (this != &that)
        {
            this->m_data = that.m_data;
        }
        return *this;
    }
    Data& operator=(Data&& that) // 移动赋值函数
    {
        std::cout << "move operator =" << std::endl;
        if (this != &that)
        {
            this->m_data = std::move(that.m_data);
        }
        return *this;
    }
private:
    std::string m_data = "there is no message";
};
 
int main()
{
    {
        Data a;
        Data b("this is a message");
        Data c = a;
        Data d = std::move(b);
        c = d;
        c = std::move(d);
        c = Data("this is a new message");
    }
    return 1;
}
