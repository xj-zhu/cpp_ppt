//2.//////////////////////////////////////////////////////
std::variant<int, float, int> a{ std::in_place_index<2>, 12 }; // 类型列表同时存在多个同类型时赋值需要指定位置
int b = std::get<2>(a); // 取值也需要按位置取值
std::variant<int, double> c = std::get<2>(a), d; // 只要用公共的类型，就可以转换
c = std::get<2>(a);
d = std::get<int>(c); // 类型列表不存在多个相同类型，可以按照类型取值
c = std::get<int>(c); // 自己给自己赋值也没毛病
