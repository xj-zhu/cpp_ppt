// c++ 11
template <typename B, typename M, typename std::enable_if<std::is_arithmetic<B>::value && std::is_arithmetic<M>::value, int>::type = 0>
auto mult(const B& b, const M& m) -> decltype(b * m)
{
    return b * m;
}
template <typename B, typename M, typename std::enable_if<std::is_convertible<B, std::string>::value && std::is_integral<M>::value, int>::type = 0>
std::string mult(const B& b, const M& m)
{
    std::string r;
    for (auto i = 0; i < m; ++i)
    {
        r += b;
    }
    return r;
}
 
// c++ 17
template <typename B, typename M>
auto mult(const B& b, const M& m)
{
    if constexpr (std::is_arithmetic<B>::value && std::is_arithmetic<M>::value)
    {
        return b * m;
    }
    if constexpr (std::is_convertible<B, std::string>::value && std::is_integral<M>::value)
    {
        std::string r;
        for (auto i = 0; i < m; ++i)
        {
            r += b;
        }
        return r;
    }
}

// 正确实现
template<typename _Tp, typename _Up>
class _is_appendable_helper
{
    template<typename _Tp1, typename _Up1,
        typename = decltype(std::declval<_Tp1>() += std::declval<_Up1>())>
        static std::true_type __test(int);
 
    template<typename, typename>
    static std::false_type __test(...);
 
public:
    typedef decltype(__test<_Tp&, _Up&>(0)) type;
};
